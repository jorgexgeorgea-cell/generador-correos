async function buscarMensajes() { if(!correoActual) return; const [user, dom] = correoActual.split('@'); const dot = document.getElementById('dot'); const status = document.getElementById('statusText'); // Intentamos con un proxy alternativo más robusto const apiURL = `https://www.1secmail.com/api/v1/?action=getMessages&login=${user}&domain=${dom}`; const proxyURL = `https://api.codetabs.com/v1/proxy?quest=${encodeURIComponent(apiURL)}`; try { const res = await fetch(proxyURL); if (!res.ok) throw new Error('Proxy fallido'); const data = await res.json(); const lista = document.getElementById('msgList'); dot.className = "h-2 w-2 bg-green-500 rounded-full mr-2 shadow-[0_0_8px_#22c55e]"; status.innerText = "Conectado"; if(data && data.length > 0) { lista.innerHTML = ""; data.forEach(m => { lista.innerHTML += `

De: ${m.from}

${m.subject}

`; }); } } catch (e) { // Si el primer proxy falla, el punto se pone rojo para avisarte dot.className = "h-2 w-2 bg-red-500 rounded-full mr-2"; status.innerText = "Reintentando conexión..."; console.error("Error detectado:", e); } }
